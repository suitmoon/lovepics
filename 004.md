<script>
    // 从纯文本标签中安全地获取数据，没有任何转义风险
    const rawQuestionsData = document.getElementById('questions-data-block').textContent;

    let stats = { answered: 0, correct: 0, wrong: 0 };

    function parseQuestions(data) {
        const questions = [];
        const blocks = data.trim().split(/\n\s*\n/);

        blocks.forEach((block, index) => {
            const lines = block.split('\n').filter(line => line.trim() !== '');
            if (lines.length < 2) return;

            // 寻找包含答案的最后一行（题干结束的地方），防止代码块中的括号被误判
            let titleEndIndex = 0;
            let answer = '';
            let titleText = '';

            for (let i = 0; i < lines.length; i++) {
                const answerMatch = lines[i].match(/[（\(]\s*([A-Z]+)\s*[）\)]\s*$/);
                // 确保它下面紧挨着的是选项 (A、 B、 C、 D、)
                if (answerMatch && i + 1 < lines.length && /^[A-Z][、\.\s]/.test(lines[i+1].trim())) {
                    answer = answerMatch[1];
                    titleEndIndex = i;
                    // 去掉题干这一行末尾的答案括号
                    lines[i] = lines[i].replace(/[（\(]\s*[A-Z]+\s*[）\)]\s*$/, '');
                    break;
                }
            }

            if (answer) {
                // 组装包含代码的多行题干
                titleText = lines.slice(0, titleEndIndex + 1).join('\n');
                
                const options = [];
                for (let i = titleEndIndex + 1; i < lines.length; i++) {
                    const optLine = lines[i].trim();
                    const optMatch = optLine.match(/^([A-Z])[、\.．\s]+(.*)$/);
                    if (optMatch) {
                        options.push({ key: optMatch[1], text: optMatch[2] });
                    }
                }

                questions.push({
                    id: index,
                    title: titleText,
                    answer: answer,
                    type: answer.length > 1 ? 'multi' : 'single',
                    options: options
                });
            }
        });
        return questions;
    }

    function renderQuiz(questions) {
        const container = document.getElementById('quiz-container');
        container.innerHTML = '';

        questions.forEach((q, qIndex) => {
            const card = document.createElement('div');
            card.className = 'question-card';
            card.id = `question-${qIndex}`;

            const typeLabel = q.type === 'single' ? '单选' : '多选';
            const inputType = q.type === 'single' ? 'radio' : 'checkbox';

            let optionsHtml = '';
            q.options.forEach(opt => {
                // 将选项中的特殊字符进行 HTML 实体转义，防止代码选项破坏页面结构
                const safeText = opt.text.replace(/</g, "&lt;").replace(/>/g, "&gt;");
                optionsHtml += `
                    <label class="option-item" id="label-${qIndex}-${opt.key}">
                        <input type="${inputType}" name="q${qIndex}" value="${opt.key}">
                        <span>${opt.key}. ${safeText}</span>
                    </label>
                `;
            });

            // 题干也做 HTML 实体转义，防止题干中的代码（如 <iostream>）变成真实的 DOM 标签
            const safeTitle = q.title.replace(/</g, "&lt;").replace(/>/g, "&gt;");

            card.innerHTML = `
                <div class="question-title">
                    <span class="question-type">${typeLabel}</span>
                    ${safeTitle}
                </div>
                <ul class="options-list">
                    ${optionsHtml}
                </ul>
                <button class="btn-submit" onclick="checkAnswer(${qIndex}, '${q.answer}')">确认答案</button>
                <div class="feedback" id="feedback-${qIndex}"></div>
            `;
            container.appendChild(card);
        });
    }

    window.checkAnswer = function(qIndex, correctAnswer) {
        const card = document.getElementById(`question-${qIndex}`);
        const inputs = card.querySelectorAll(`input[name="q${qIndex}"]`);
        const feedback = document.getElementById(`feedback-${qIndex}`);
        const submitBtn = card.querySelector('.btn-submit');

        let selectedAns = [];
        inputs.forEach(input => { if (input.checked) selectedAns.push(input.value); });

        const selectedStr = selectedAns.sort().join('');
        const correctStr = correctAnswer.split('').sort().join('');

        if (selectedStr === '') { alert('请先选择答案！'); return; }

        inputs.forEach(input => input.disabled = true);
        submitBtn.disabled = true;
        stats.answered++;

        if (selectedStr === correctStr) {
            stats.correct++;
            feedback.innerHTML = '<span style="color: var(--success-color);">回答正确！</span>';
            selectedAns.forEach(key => document.getElementById(`label-${qIndex}-${key}`).classList.add('correct-answer'));
        } else {
            stats.wrong++;
            feedback.innerHTML = `<span style="color: var(--error-color);">回答错误。</span> 正确答案是：${correctAnswer}`;
            inputs.forEach(input => {
                const label = document.getElementById(`label-${qIndex}-${input.value}`);
                if (correctStr.includes(input.value)) label.classList.add('correct-answer');
                else if (input.checked) label.classList.add('wrong-answer');
            });
        }
        feedback.style.display = 'block';
        updateStatsUI();
    };

    function updateStatsUI() {
        document.getElementById('total-answered').innerText = stats.answered;
        document.getElementById('total-correct').innerText = stats.correct;
        document.getElementById('total-wrong').innerText = stats.wrong;
    }

    function init() {
        const questions = parseQuestions(rawQuestionsData);
        if(questions.length > 0) renderQuiz(questions);
        else document.getElementById('quiz-container').innerHTML = '<p>暂未解析到题目，请检查数据格式。</p>';
    }

    init();
</script>
