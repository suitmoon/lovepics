import re

def verify_one_group(grep_ret):
    cache = {}
    # 匹配模型名、动作(Load/Unload)、cnt值
    pattern = r"Stat for ([\w#]+): \[(Load|Unload)\].*?cnt:(\d+)"

    for line in grep_ret.split('\n'):
        match = re.search(pattern, line)
        if not match:
            continue
            
        name, action, cnt = match.groups()
        cnt = int(cnt)

        # 按模型名称存放
        if name not in cache:
            cache[name] = {}
        cache[name][action] = cnt

        # 只要凑齐一对，立刻比较并返回结果，不再看后面的日志
        if "Load" in cache[name] and "Unload" in cache[name]:
            # 返回 True (满足要求) 或 False (不满足)
            return cache[name]["Load"] <= cache[name]["Unload"]

    return None # 如果整段日志都没凑齐任何一对，返回 None
