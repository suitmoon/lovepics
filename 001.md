        int head = 0,tail = nums.size()-1;
        int flag = tail;
        vector<int> res(flag,0);
        while(head<tail){
            int a= nums[head];
            int b = nums[tail];
            if(a*a<=b*b){res[flag--]=b*b;tail--;continue;}
            else{
                res[flag--]=a*a;head--;continue;
            }
        }
        res[0] = nums[head]*nums[head];
        return res;
