class Solution {
public:
    ListNode* remove(ListNode* head, int& n,int target) {
        if(!head||!head->next){
          n++;
          return head;
        }
        auto node = remove(head->next,n,target);
        if(n==target){
          head->next = head->next->next;
        }else{
          n++;
        }
        return node;
    }
    ListNode* removeNthFromEnd(ListNode* head, int n) {
        ListNode node;
        node.next = head;
        int m = 0;
        remove(&node,m,n+1);
        return node.next;
    }
};
